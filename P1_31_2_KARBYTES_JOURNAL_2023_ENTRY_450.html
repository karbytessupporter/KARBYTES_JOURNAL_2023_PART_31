Write a C++ function which shows how to algorithmically compute the natural log of some number.

#include < iostream >
#include < cmath >
using namespace std;

// This function uses an algorithm to calculate the natural log of a number
double naturalLog(double value)
{
    double result = 0.0; // This value will hold the natural log result

    // Calculate the natural log using a Taylor Series.
    // The series is truncated after 15 terms.
    for(int i = 1; i <= 15; i++)
    {
        double numerator = pow(-1, i - 1) * pow(value - 1, i);
        double denominator = i * pow(value, i);
        result += numerator / denominator; 
    }

    // Return the result
    return result;
}

int main()
{
    double number = 7.4; // Sample input number
    double logResult = naturalLog(number);
    cout << "The natural log of " << number << " is " << logResult << endl;
    return 0;
}
